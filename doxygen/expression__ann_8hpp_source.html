180  template <typename U, enable_double<U> = 0>
181  std::tuple<U, std::vector<U>, std::vector<U>> d_mse(const std::vector<U> &point, const std::vector<U> &prediction)
182  {
183  if (point.size() != this->get_n()) {
184  throw std::invalid_argument("When computing the mse the point dimension (input) seemed wrong, it was: "
185  + std::to_string(point.size())
186  + " while I expected: " + std::to_string(this->get_n()));
187  }
188  if (prediction.size() != this->get_m()) {
189  throw std::invalid_argument(
190  "When computing the mse the prediction dimension (output) seemed wrong, it was: "
191  + std::to_string(prediction.size()) + " while I expected: " + std::to_string(this->get_m()));
192  }
193  U value(U(0.));
194  std::vector<U> gweights(m_weights.size(), U(0.));
195  std::vector<U> gbiases(m_biases.size(), U(0.));
196 
197  // ------------------------------------------ Forward pass (takes roughly half of the time) --------------------
198  // All active nodes outputs get computed as well as
199  // the activation function derivatives
200  auto n_nodes = this->get_n() + this->get_rows() * this->get_cols();
201  std::vector<U> node(n_nodes, 0.), d_node(n_nodes, 0.);
202  fill_nodes(point, node, d_node); // here is where the computations happen.
203 
204  // We add to node and node_d some virtual nodes containing the output values (x_i-\hat x_i) ^ 2, and its
205  // derivative 2(x_i - \hat x_i) so that the acyclic graph is now computing all blocks of the mse (a dCGP-ANN
206  // only computes outputs)
207  for (decltype(this->get_m()) i = 0u; i < this->get_m(); ++i) {
208  auto node_idx = this->get()[this->get().size() - this->get_m() + i];
209  auto dummy = (node[node_idx] - prediction[i]);
210  node.push_back(dummy * dummy);
211  d_node.push_back(2 * dummy);
212  value += dummy * dummy;
213  }
214 
215  // ------------------------------------------ Backward pass (takes roughly the remaining half) -----------------
216  // We iterate backward on all the active nodes (except the input nodes)
217  // filling up the gradient information at each node for the incoming weights and relative bias
218  for (auto it = this->get_active_nodes().rbegin(); it != this->get_active_nodes().rend(); ++it) {
219  if (*it < this->get_n()) continue;
220  // index of the node in the bias vector
221  auto b_idx = *it - this->get_n();
222  // index of the node in the weight vector
223  auto w_idx = this->get_arity() * (*it - this->get_n());
224  // index of the node in the chromosome
225  auto c_idx = (*it - this->get_n()) * (this->get_arity() + 1);
226  // index in the node/d_node vectors
227  auto n_idx = *it;
228  // We update the d_node information
229  U cum = 0.;
230  for (auto i = 0u; i < m_connected[*it].size(); ++i) {
231  // If the node is not "virtual", that is not one of the m virtual nodes we added computing (x-x_i)^2
232  if (m_connected[*it][i].first < this->get_n() + this->get_rows() * this->get_cols()) {
233  cum += m_weights[m_connected[*it][i].second] * d_node[m_connected[*it][i].first];
234  } else {
235  auto n_out = m_connected[*it][i].first - (this->get_n() + this->get_rows() * this->get_cols());
236  cum += d_node[d_node.size() - this->get_m() + n_out];
237  }
238  }
239  d_node[n_idx] *= cum;
240 
241  // fill gradients for weights and biases info
242  for (auto i = 0u; i < this->get_arity(); ++i) {
243  gweights[w_idx + i] = d_node[n_idx] * node[this->get()[c_idx + 1 + i]];
244  }
245  gbiases[b_idx] = d_node[n_idx];
246  }
247 
248  return std::make_tuple(std::move(value), std::move(gweights), std::move(gbiases));
249  }

286  template <typename U, enable_double<U> = 0>
287  void sgd(const std::vector<std::vector<U>> &points, const std::vector<std::vector<U>> &predictions, double l_rate,
288  unsigned batch_size)
289  {
290  if (points.size() != predictions.size()) {
291  throw std::invalid_argument("Data and label size mismatch data size is: " + std::to_string(points.size())
292  + " while label size is: " + std::to_string(predictions.size()));
293  }
294  if (points.size() == 0) {
295  throw std::invalid_argument("Data size cannot be zero");
296  }
297  if (l_rate <= 0) {
298  throw std::invalid_argument("The learning rate must be a positive number, while: " + std::to_string(l_rate)
299  + " was detected.");
300  }
301  auto dfirst = points.begin();
302  auto dlast = points.end();
303  auto lfirst = predictions.begin();
304  while (dfirst != dlast) {
305  if (dfirst + batch_size > dlast) {
306  update_weights<U>(dfirst, dlast, lfirst, l_rate);
307  dfirst = dlast;
308  } else {
309  update_weights<U>(dfirst, dfirst + batch_size, lfirst, l_rate);
310  dfirst += batch_size;
311  lfirst += batch_size;
312  }
313  }

569  // For numeric computations
570  template <typename U, enable_double<U> = 0>
571  U kernel_call(std::vector<U> &function_in, unsigned idx, unsigned weight_idx, unsigned bias_idx) const
572  {
573  // Weights (we transform the inputs a,b,c,d,e in w_1 a, w_2 b, w_3 c, etc...)
574  for (auto j = 0u; j < this->get_arity(); ++j) {
575  function_in[j] = function_in[j] * m_weights[weight_idx + j];
576  }
577  // Biases (we add to the first input a bias so that a,b,c,d,e goes in c, etc...))
578  function_in[0] += m_biases[bias_idx];
579  // We compute the node function that will, for example, map w_1 a + bias, w_2 b, w_3 c,... into f(w_1 a +
580  // w_2 b + w_3 c + ... + bias)
581  return this->get_f()[this->get()[idx]](function_in);
582  }

584  // For the symbolic expression
585  template <typename U, typename std::enable_if<std::is_same<U, std::string>::value, int>::type = 0>
586  U kernel_call(std::vector<U> &function_in, unsigned idx, unsigned weight_idx, unsigned bias_idx) const
587  {
588  // Weights
589  for (auto j = 0u; j < this->get_arity(); ++j) {
590  function_in[j] = m_weights_symbols[weight_idx + j] + "*" + function_in[j];
591  }
592  // Biases
593  function_in[0] = m_biases_symbols[bias_idx] + "+" + function_in[0];
594  return this->get_f()[this->get()[idx]](function_in);
595  }

597  // computes node to evaluate the expression
598  template <typename U, enable_double_string<U> = 0>
599  std::vector<U> fill_nodes(const std::vector<U> &in) const
600  {
601  if (in.size() != this->get_n()) {
602  throw std::invalid_argument("Input size is incompatible");
603  }
604  std::vector<U> node(this->get_n() + this->get_rows() * this->get_cols());
605  std::vector<U> function_in(this->get_arity());
606  for (auto i : this->get_active_nodes()) {
607  if (i < this->get_n()) {
608  node[i] = in[i];
609  } else {
610  // position in the chromosome of the current node
611  unsigned idx = (i - this->get_n()) * (this->get_arity() + 1);
612  // starting position in m_weights of the weights relative to the node
613  unsigned weight_idx = (i - this->get_n()) * this->get_arity();
614  // starting position in m_biases of the node bias
615  unsigned bias_idx = i - this->get_n();
616  for (auto j = 0u; j < this->get_arity(); ++j) {
617  function_in[j] = node[this->get()[idx + j + 1]];
618  }
619  node[i] = kernel_call(function_in, idx, weight_idx, bias_idx);
620  }
621  }
622  return node;
623  }

625  // computes node and node_d to start backprop
626  template <typename U, enable_double<U> = 0>
627  void fill_nodes(const std::vector<U> &in, std::vector<U> &node, std::vector<U> &d_node) const
628  {
629  if (in.size() != this->get_n()) {
630  throw std::invalid_argument("Input size is incompatible");
631  }
632  // Start
633  std::vector<U> function_in(this->get_arity());
634  for (auto i : this->get_active_nodes()) {
635  if (i < this->get_n()) {
636  node[i] = in[i];
637  // We need d_node to have the same structure of node, hence we also
638  // put some bogus entries fot the input nodes that actually do not have an activation function
639  // hence no need/use/meaning for a derivative
640  d_node[i] = 0.;
641  } else {
642  // position in the chromosome of the current node
643  unsigned idx = (i - this->get_n()) * (this->get_arity() + 1);
644  // starting position in m_weights of the weights relative to the node
645  unsigned weight_idx = (i - this->get_n()) * this->get_arity();
646  // starting position in m_biases of the node bias
647  unsigned bias_idx = i - this->get_n();
648  for (auto j = 0u; j < this->get_arity(); ++j) {
649  function_in[j] = node[this->get()[idx + j + 1]];
650  }
651  node[i] = kernel_call(function_in, idx, weight_idx, bias_idx);
652  // take cares of d_node
653  // sigmoid derivative is sig(1-sig)
654  if (this->get_f()[this->get()[idx]].get_name() == "sig") {
655  d_node[i] = node[i] * (1. - node[i]);
656  // tanh derivative is 1 - tanh**2
657  } else if (this->get_f()[this->get()[idx]].get_name() == "tanh") {
658  d_node[i] = 1. - node[i] * node[i];
659  // Relu derivative is 0 if relu<0, 1 otherwise
660  } else if (this->get_f()[this->get()[idx]].get_name() == "ReLu") {
661  d_node[i] = (node[i] > 0.) ? 1. : 0.;
662  }
663  }
664  }
665  return;
666  }

668  // This overrides the base class update_data_structures and updates also the m_connected (as well as
669  // m_active_nodes and genes). It is called upon construction and each time active genes are changed.
670  void update_data_structures()
671  {
672  expression<T>::update_data_structures();
673  m_connected.clear();
674  m_connected.resize(this->get_n() + this->get_m() + this->get_rows() * this->get_cols());
675  for (auto node_id : this->get_active_nodes()) {
676  if (node_id >= this->get_n()) { // not for input nodes
677  // position in the chromosome of the current node
678  unsigned idx = (node_id - this->get_n()) * (this->get_arity() + 1);
679  // loop over the genes representing connections
680  for (auto i = idx + 1; i < idx + 1 + this->get_arity(); ++i) {
681  if (this->is_active(this->get()[i])) {
682  m_connected[this->get()[i]].push_back(
683  {node_id, (node_id - this->get_n()) * this->get_arity() + i - idx - 1});
684  }
685  }
686  }
687  }
688  // We now add the output nodes with ids starting from n + r * c. In this case the weight is not
689  // relevant, hence we use the arbitrary value 0u as index in the weight vector.
690  for (auto i = 0u; i < this->get_m(); ++i) {
691  auto virtual_idx = this->get_n() + this->get_rows() * this->get_cols() + i;
692  auto node_idx = this->get()[this->get().size() - this->get_m() + i];
693  m_connected[node_idx].push_back({virtual_idx, 0u});
694  }
695  }

709  template <typename U, enable_double<U> = 0>
710  void update_weights(typename std::vector<std::vector<U>>::const_iterator dfirst,
711  typename std::vector<std::vector<U>>::const_iterator dlast,
712  typename std::vector<std::vector<U>>::const_iterator lfirst, U lr)
713  {
714  U coeff(lr / static_cast<U>(dlast - dfirst));
715  while (dfirst != dlast) {
716  auto mse_out = d_mse(*dfirst++, *lfirst++);
717  std::transform(m_weights.begin(), m_weights.end(), std::get<1>(mse_out).begin(), m_weights.begin(),
718  [coeff](U a, U b) { return a - coeff * b; });
719  std::transform(m_biases.begin(), m_biases.end(), std::get<2>(mse_out).begin(), m_biases.begin(),
720  [coeff](U a, U b) { return a - coeff * b; });
721  }
722  }

724  template <typename U, enable_double<U> = 0>
725  std::tuple<U, std::vector<U>, std::vector<U>> d_mse(typename std::vector<std::vector<U>>::const_iterator dfirst,
726  typename std::vector<std::vector<U>>::const_iterator dlast,
727  typename std::vector<std::vector<U>>::const_iterator lfirst)
728  {
729  U value(U(0.));
730  std::vector<U> gweights(m_weights.size(), U(0.));
731  std::vector<U> gbiases(m_biases.size(), U(0.));
732  U dim = static_cast<U>(dlast - dfirst);
733 
734  while (dfirst != dlast) {
