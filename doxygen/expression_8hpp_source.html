572  virtual void update_data_structures()
573  {
574  assert(m_x.size() == m_lb.size());
575 
576  // First we update the active nodes
577  std::vector<unsigned> current(m_m), next;
578  m_active_nodes.clear();
579  // At the beginning, current contains only the output nodes connections
580  for (auto i = 0u; i < m_m; ++i) {
581  current[i] = m_x[(m_arity + 1) * m_r * m_c + i];
582  }
583  do {
584  m_active_nodes.insert(m_active_nodes.end(), current.begin(), current.end());
585 
586  for (auto node_id : current) {
587  if (node_id >= m_n) // we insert the input nodes connections as they do
588  // not have any
589  {
590  for (auto i = 1u; i <= m_arity; ++i) {
591  next.push_back(m_x[(node_id - m_n) * (m_arity + 1) + i]);
592  }
593  } else {
594  m_active_nodes.push_back(node_id);
595  }
596  }
597  // We remove duplicates to avoid processing them and thus having a 2^N
598  // complexity
599  std::sort(next.begin(), next.end());
600  next.erase(std::unique(next.begin(), next.end()), next.end());
601  current = next;
602  next.clear();
603  } while (current.size() > 0);
604 
605  // We remove duplicates and keep m_active_nodes sorted
606  std::sort(m_active_nodes.begin(), m_active_nodes.end());
607  m_active_nodes.erase(std::unique(m_active_nodes.begin(), m_active_nodes.end()), m_active_nodes.end());
608 
609  // Then the active genes
610  m_active_genes.clear();
611  for (auto i = 0u; i < m_active_nodes.size(); ++i) {
612  if (m_active_nodes[i] >= m_n) {
613  unsigned idx = (m_active_nodes[i] - m_n) * (m_arity + 1);
614  for (auto j = 0u; j <= m_arity; ++j) {
615  m_active_genes.push_back(idx + j);
616  }
617  }
618  }
619  for (auto i = 0u; i < m_m; ++i) {
620  m_active_genes.push_back(m_r * m_c * (m_arity + 1) + i);
621  }
622  }
